### My Solution

	public int firstUniqChar(String s) {
		for(int i=0;i<s.length();i++){
				String fetchOneChar = s.substring(i,i+1);
				int b = s.indexOf(fetchOneChar);
				int e = s.lastIndexOf(fetchOneChar);
				if(i == b && b == e){
					return i;
				}
			}
			return -1;
	}

	public class Solution {
		public int firstUniqChar(String s) {
			int freq [] = new int[26];
			for(int i = 0; i < s.length(); i ++)
				freq [s.charAt(i) - 'a'] ++;
			for(int i = 0; i < s.length(); i ++)
				if(freq [s.charAt(i) - 'a'] == 1)
					return i;
			return -1;
		}
